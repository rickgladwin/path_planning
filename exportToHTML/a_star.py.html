1    """ 
2    Use the A* algorithm to solve this 8-puzzle. 
3     
4    ┌───────────────────────┐       ┌───────────────────────┐ 
5    │      Initial State    │       │       Goal State      │ 
6    │                       │       │                       │ 
7    ├───────┬───────┬───────┤       ├───────┬───────┬───────┤ 
8    │   2   │   8   │   3   │       │   1   │   2   │   3   │ 
9    ├───────┼───────┼───────┤       ├───────┼───────┼───────┤ 
10   │   1   │   6   │   4   │       │   8   │       │   4   │ 
11   ├───────┼───────┼───────┤       ├───────┼───────┼───────┤ 
12   │   7   │       │   5   │       │   7   │   6   │   5   │ 
13   └───────┴───────┴───────┘       └───────┴───────┴───────┘ 
14    
15   https://monosketch.io/ 
16    
17   The simple evaluation function f(x) is defined as follows: 
18   f(x) = g(x)+h(x) 
19   where, 
20   g(x) = depth of node X in the search tree (equivalent to cost so far) 
21   h(x) = the number of tiles not in their goal position in a given state X (equivalent to minimum cost to completion) 
22    
23   States: Location of eight tiles plus blank 
24   Initial state: Any state can be designated as the initial state 
25   Actions: Blank tile moves left, right, up, or down and exchanges places with a numbered tile 
26   Successor function: Actions have their expected effects 
27   Goal test: Check whether the goal configuration is reached 
28   Path cost: Each step costs 1 
29    
30   Determine the following: 
31   - Is this puzzle solvable? 
32   - If so, what is the minimum number of steps to complete the puzzle? 
33    
34   Extended challenge: Write a Python script to solve the puzzle using the A* algorithm. 
35   Whether you feel you have the skills to do this or not, it is a valuable exercise to work through the steps required. 
36    
37   ---- 
38    
39   - branching factor will be minimum 2, maximum 4 for any given state, depending on the location of the blank space. 
40   - it should be possible to generate a state graph for this problem by applying the valid actions 
41     to any state on the frontier. 
42    
43   """

46   # in classes that are the type of the class itself)

52   class Problem:
53       """ 
54       The properties and constraints on the problem, and the actions and constraints on the solution(s) 
55       """
56       start_node: StateNode
57       goal_node: StateNode
58       actions: tuple
59   
60       def __init__(self, start_node, goal_node):
61           self.start_node = start_node
62           self.goal_node = goal_node
63           self.actions = (self.swap_up, self.swap_down, self.swap_right, self.swap_left)
64   
65       """ 
66       The actions for the problem that move from one SearchNode to another 
67       are a set of swaps between the blank space and another tile. 
68        
69       As these actions are intended to generate new SearchNodes and not just 
70       move between known states, the functions will return StateNodes, which 
71       we will then use to generate new SearchNodes. 
72       """
73       def swap_up(self, from_state: StateNode.state_id) -> StateNode.state_id:
74           """ 
75           012 --> no swaps 
76           345 --> swap index with -3 
77           678 --> swap index with -3 
78           """
79           new_state = from_state
80           blank_index = from_state.index("0")
81   
82           if blank_index <= 2:
83               raise RuntimeError("Can't swap up: blank is already in top row.")
84   
85           swap_index = blank_index - 3
86           swap_tile = from_state[swap_index]
87           new_state = str_replace_char_at(new_state, blank_index, swap_tile)
88           new_state = str_replace_char_at(new_state, swap_index, "0")
89   
90           return new_state
91   
92   
93       def swap_down(self, from_state: StateNode.state_id) -> StateNode.state_id:
94           """ 
95           012 --> swap index with +3 
96           345 --> swap index with +3 
97           678 --> no swaps 
98           """
99           new_state = from_state
100          blank_index = from_state.index("0")
101  
102          if blank_index >= 6:
103              raise RuntimeError("Can't swap down: blank is already in bottom row.")
104  
105          swap_index = blank_index + 3
106          swap_tile = from_state[swap_index]
107          new_state = str_replace_char_at(new_state, blank_index, swap_tile)
108          new_state = str_replace_char_at(new_state, swap_index, "0")
109  
110          return new_state
111  
112  
113      def swap_right(self, from_state: StateNode.state_id) -> StateNode.state_id:
114          """ 
115          012 
116          345 
117          678 
118            ↑ no swaps 
119          ↑↑  index + 1 
120          """
121          new_state = from_state
122          blank_index = from_state.index("0")
123  
124          if blank_index in (2,5,8):
125              raise RuntimeError("Can't swap right: blank is already in right column.")
126  
127          swap_index = blank_index + 1
128          swap_tile = from_state[swap_index]
129          new_state = str_replace_char_at(new_state, blank_index, swap_tile)
130          new_state = str_replace_char_at(new_state, swap_index, "0")
131  
132          return new_state
133  
134  
135      def swap_left(self, from_state: StateNode.state_id) -> StateNode.state_id:
136          """ 
137          012 
138          345 
139          678 
140          ↑   no swaps 
141           ↑↑ index - 1 
142          """
143          new_state = from_state
144          blank_index = from_state.index("0")
145  
146          if blank_index in (0,3,6):
147              raise RuntimeError("Can't swap right: blank is already in left column.")
148  
149          swap_index = blank_index - 1
150          swap_tile = from_state[swap_index]
151          new_state = str_replace_char_at(new_state, blank_index, swap_tile)
152          new_state = str_replace_char_at(new_state, swap_index, "0")
153  
154          return new_state

157  def str_replace_char_at(string: str, char_index: int, new_char: chr) -> str:
158      if char_index < 0 or char_index >= len(string):
159          raise ValueError("position must be within string length")
160  
161      new_string = string[:char_index] + new_char + string[char_index + 1:]
162      return new_string

165  class StateNode:
166      """ 
167      A Node in the underlying state data from the problem. 
168      """
169      # a minimal representation of the state of the puzzle board, e.g.
170      # ├─┬─┬─┤
171      # │2│8│3│
172      # ├─┼─┼─┤
173      # │1│6│4│ ==> "283164705"
174      # ├─┼─┼─┤
175      # │7│ │5│
176      # └─┴─┴─┘
177      #
178      state_id: str
179  
180      # NOTE: in this case, the actions available from each state will follow a set
181      #  of rules that will depend on the position of the blank square. So we can
182      #  assign responsibility for this information to a more singular entity,
183      #  like a Problem class, a Solver/Traverser class, or even the Frontier class, as part
184      #  of the expand() function (though this seems like a stretch – the expand() function
185      #  should expand based on what it's given, not be responsible for knowing the rules of expansion)
186  
187      # NOTE: the fact that this class consists of a single property and a generic constructor
188      #  is a code smell, but it's a result of trying to build these search solvers in the most
189      #  general way possible, ideally being able to refactor the general structure in a simple way
190      #  in order to adapt the template to different algorithms and problem types.
191  
192      def __init__(self, state_id):
193          self.state_id = state_id
194

196  class SearchNode:
197      """ 
198      A Node in the search graph, generated while solving the problem. 
199      """
200      problem: Problem
201      state: StateNode
202      parent: SearchNode
203      path_cost: int
204      actions: tuple
205  
206      def __init__(self, problem: Problem, state_node: StateNode, parent_node: SearchNode | None, path_cost: int | float):
207          self.problem = problem
208          self.state: StateNode = state_node
209          self.parent: SearchNode = parent_node
210          # the action that led to this SearchNode
211          # self.action = action
212          # set the actions available to this SearchNode
213          # NOTE: setting the actions functions as properties on the SearchNodes is sensible so long as
214          #  a) memory isn't at risk of being used up and
215          #  b) the functions are passed by reference and stored as pointers until they're actually used
216          self.set_actions()
217          self.path_cost = path_cost
218  
219      def set_actions(self) -> None:
220          """ 
221          Based on the state, some actions are permitted and some are excluded, depending on the 
222          position of the blank space. Set the available actions for this SearchNode. 
223          """
224          blank_index = self.state.state_id.index("0")
225          node_actions = list(self.problem.actions)
226          if blank_index <= 2:
227              # can't swap up
228              node_actions.remove(self.problem.actions[0])
229          if blank_index >= 6:
230              # can't swap down
231              node_actions.remove(self.problem.actions[1])
232          if blank_index in (2,5,8):
233              # can't swap right
234              node_actions.remove(self.problem.actions[2])
235          if blank_index in (0,3,6):
236              # can't swap left
237              node_actions.remove(self.problem.actions[3])
238          self.actions = tuple(node_actions)

241  class Frontier:
242      """ 
243      The Frontier class for best-first-search. 
244      For A*, evaluate() uses some metric for judging the cost for a node in the frontier, 
245      for example ortholinear proximity to the goal node. For an 8-puzzle, we're using 
246      f(x) = g(x)+h(x) 
247      where 
248      g(x) = depth of node X in the search tree (equivalent to cost so far) 
249      h(x) = the number of tiles not in their goal position in a given state X (equivalent to minimum cost to completion) 
250      Note that we could use the sum of the distances of each tile from the goal state instead of/in addition to h(x). 
251      The exact evaluation function alters the algorithm, but is only used to select one node from the frontier, so it 
252      only needs to be as fine-tuned as that task requires. 
253      """
254      nodes: set[SearchNode]
255      goal_node: SearchNode
256  
257      def __init__(self, goal_node):
258          self.nodes = set()
259          self.goal_node = goal_node
260  
261      def is_empty(self):
262          return not self.nodes
263  
264      def pop(self) -> SearchNode | None:
265          # return the node in the frontier with the best score based on the evaluation function
266          # and remove it from the set.
267          top_node = self.top()
268          self.nodes.remove(top_node)
269          return top_node
270  
271      def top(self) -> SearchNode | None:
272          # return the node in the frontier with the best score (lowest cost, typically)
273          # based on the evaluation function, but do not remove it from the set.
274          if not self.nodes:
275              return None
276          best_node = next(iter(self.nodes))
277          for node in self.nodes:
278              if self.evaluate(node) < best_node.path_cost:
279                  best_node = node
280          return best_node
281  
282      def add(self, node: SearchNode) -> None:
283          # add the node to the set (unordered, in this example)
284          self.nodes.add(node)
285          # print(f"Frontier nodes: {self}")
286  
287      def evaluate(self, node: SearchNode) -> int:
288          # The evaluation function f(n).
289          # return a score for the node based on the criteria defined by the problem or algorithm type
290          return node.path_cost + self.nodes_out_of_goal_count(node)
291  
292      def nodes_out_of_goal_count(self, node: SearchNode) -> int:
293          """ 
294          e.g.                           ↓↓  ↓ ↓↓ 
295          search_node.state.state_id = [234567018] 
296          goal_node.state.state_id   = [243560781] 
297          count = 5 (we're not counting the blank space) 
298          """
299          search_state = node.state.state_id
300          goal_state   = self.goal_node.state.state_id
301          count = 0
302          for i in range(0, len(goal_state)):
303              if goal_state[i] != 0 and goal_state[i] != search_state[i]:
304                  count += 1
305          return count
306  
307      def __str__(self):
308          output = ""
309          for node in self.nodes:
310              output += f"{node.state.state_id}, cost: {node.path_cost}\n"
311          return output

314  class Solver:
315      problem: Problem
316      frontier: Frontier # nodes in the search graph that have been generated but not expanded (visited)
317      reached: dict[Hashable, SearchNode] # SearchNodes that have been generated or expanded,
318      # i.e. nodes in the frontier plus nodes that have been visited
319      # NOTE: (expanded nodes) ∪ (frontier nodes) == reached nodes
320      start_node: SearchNode
321      goal_node: SearchNode
322  
323      def __init__(self, problem: Problem):
324          self.problem = problem
325          self.start_node = SearchNode(self.problem, problem.start_node, None, 0)
326          self.goal_node = SearchNode(self.problem, problem.goal_node, None, float('inf'))
327          self.frontier = Frontier(self.goal_node)
328          self.frontier.add(self.start_node)
329          self.reached = {self.start_node.state.state_id: self.start_node}
330  
331      def solve(self) -> None:
332          while not self.frontier.is_empty():
333              # visit phase
334              current_node = self.frontier.pop()
335              if current_node.state.state_id == self.goal_node.state.state_id:
336                  self.finish(True, current_node)
337                  return
338              # expand phase
339              for child_node in self.expand(current_node):
340                  if (child_node.state.state_id not in self.reached.keys()
341                          or child_node.path_cost < self.reached[child_node.state.state_id].path_cost):
342                      self.reached[child_node.state.state_id] = child_node
343                      self.frontier.add(child_node)
344          self.finish(False, None)
345  
346      def expand(self, node: SearchNode) -> set[SearchNode]:
347          expanded: set = set()
348          for action in node.actions:
349              path_cost = node.path_cost + 1
350              child_state_id = action(node.state.state_id)
351              child_node = SearchNode(self.problem, StateNode(child_state_id), node, path_cost)
352              expanded.add(child_node)
353          return expanded
354  
355      def finish(self, success: bool, last_node: SearchNode | None):
356          if success:
357              print(f'success!')
358              print(f'reached goal node: {last_node.state.state_id}')
359              print(f"goal node cost: {last_node.path_cost}")
360              solution_path: list[Hashable] = []
361              path_node = last_node
362              reached_start = False
363              while not reached_start:
364                  solution_path.append(path_node.state.state_id)
365                  if not path_node.parent:
366                      reached_start = True
367                  else:
368                      path_node = path_node.parent
369              solution_path.reverse()
370              print(f"path to goal: {solution_path}")
371          else:
372              print(f'finished without success.')
373              print(f"reached nodes:")
374              for key, value in self.reached.items():
375                  print(f"{key}, cost: {value.path_cost}")
