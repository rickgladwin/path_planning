78   class StateNode:
79       """ 
80       A Node in the underlying state data from the problem. 
81       For any problem for which we can draw a state diagram, there will be equivalency 
82       between the graph of StateNodes and the graph of SearchNodes, but the properties 
83       on the nodes and the configuration of the edges may differ. 
84    
85       In this example, StateNodes will correspond directly to the nodes in the networkx graph for 
86       the problem. 
87       """
88       state: Hashable
89       # edges type is a concretion of Graph.adj[_Node] type dict[Hashable | Any, dict[str, Any]]
90       # that lets us use string node identifiers and { 'weight': int } dictionary entries for edge properties
91       edges: EdgeDataView
92   
93       def __init__(self, state, edges):
94           self.state = state
95           self.edges = edges
96   
97   class SearchNode:
98       """ 
99       A Node in the search graph, generated while solving the problem. 
100      Each SearchNode may correspond to each StateNode in the underlying data 1-to-1, 
101      especially if the underlying problem can be formulated using a state diagram. 
102      A SearchNode may have a similar relationship to the search graph as its corresponding 
103      StateNode has to the data graph but with different properties (relevant to the 
104      search), and the structure of the search graph may not match the state graph. 
105      """
106      def __init__(self, state_node: StateNode, parent_node: SearchNode | None, path_cost: int | float):
107          self.state: StateNode = state_node
108          self.parent: SearchNode = parent_node
109          # Note: for Dijkstra's, we're letting the information about the actions belong to the
110          #  expansion function, since it's straightforward (expand from a parent node, add the
111          #  parent node's path cost to the edge cost from the parent to the current node to get
112          #  the current node's path cost)
113          # self.action = action
114          self.path_cost = path_cost
115  
116  
117  class Frontier:
118      """ 
119      The Frontier class for best-first-search. 
120      dijkstra.Frontier uses smallest path size for evaluation 
121      and (in this example), a set (which is unordered) for storing nodes. 
122      NOTE: in the networkx package, the type of a Node can be any hashable 
123      type, and it's more messy than it's worth (and at the cost of polymorphism) 
124      to restrict the type of the node itself in the classes in this module for 
125      the *state* nodes (the nodes in the underlying data from the problem). 
126      For state nodes, It's enough to declare the types of the Graphs (etc.) 
127      themselves, and let the requirements from the networkx package trigger 
128      warnings while using those types. 
129      For *search* nodes, which may contain additional data, and which represent 
130      parts of the search graph (which may not even be the same shape as the 
131      underlying data graph), we can declare any type that suits us, including 
132      custom types. 
133      """
134      nodes: set[SearchNode]
135  
136      def __init__(self):
137          self.nodes = set()
138  
139      def is_empty(self):
140          return not self.nodes
141  
142      def pop(self) -> SearchNode | None:
143          # return the node in the frontier with the best score based on the evaluation function
144          # and remove it from the set.
145          top_node = self.top()
146          self.nodes.remove(top_node)
147          return top_node
148  
149      def top(self) -> SearchNode | None:
150          # return the node in the frontier with the best score (lowest cost, in the case of Dijkstra's
151          # algorithm) based on the evaluation function, but do not remove it from the set.
152          best_node = next(iter(self.nodes))
153          for node in self.nodes:
154              if best_node and self.evaluate(node) < best_node.path_cost:
155                  best_node = node
156          return best_node
157  
158      def add(self, node: SearchNode) -> None:
159          # add the node to the set (unordered, in this example)
160          self.nodes.add(node)
161          print(f"Frontier nodes: {self}")
162  
163      def evaluate(self, node: SearchNode) -> int:
164          # The evaluation function f(n).
165          # return a score for the node based on the criteria defined by the problem or algorithm type
166          return node.path_cost
167  
168      def __str__(self):
169          output = ""
170          for node in self.nodes:
171              output += f"{node.state.state}, cost: {node.path_cost}\n"
172          return output

175  class Traverser:
176      # NOTE: in the visitor pattern, the original data is kept minimal
177      #  and immutable (immutable by the visitor anyway), and an internal
178      #  representation of the data and search results is built that belongs
179      #  to the visitor. The way these search-based solutions work essentially
180      #  implements the visitor pattern by default, and runs with it by constructing
181      #  a whole search graph based on the problem data.
182      graph: Graph # original (immutable) graph data
183      frontier: Frontier # nodes in the search graph that have been generated but not expanded (visited)
184      reached: dict[Hashable, SearchNode] # SearchNodes that have been generated or expanded, i.e. nodes in the frontier
185                                # plus nodes that have been visited
186      # NOTE: (expanded nodes) ∪ (frontier nodes) == reached nodes
187      start_node: SearchNode
188      goal_node: SearchNode
189  
190      def __init__(self, graph: Graph, start_node_id: Hashable, goal_node_id: Hashable):
191          # NOTE: for Dijkstra's algorithm, the Graph (including weighted edges), start_node,
192          #  and goal_node comprise the problem, alongside the assumption that every action
193          #  involves traversing from one node to an adjacent node, and the action cost is
194          #  the weight of the edge, and that the optimal solution involves the path with
195          #  lowest cost.
196          self.graph = graph
197          self.start_node = SearchNode(StateNode(start_node_id, graph.edges(start_node_id, data=True)), None, 0)
198          self.goal_node = SearchNode(StateNode(goal_node_id, graph.edges(goal_node_id, data=True)), None, float('inf'))
199          self.frontier = Frontier()
200          self.frontier.add(self.start_node)
201          self.reached = {self.start_node.state.state: self.start_node}
202  
203      def solve(self) -> None:
204          while not self.frontier.is_empty():
205              # visit phase
206              current_node = self.frontier.pop()
207              if current_node.state.state == self.goal_node.state.state:
208                  self.finish(True, current_node)
209                  return
210              # expand phase
211              for child_node in self.expand(current_node):
212                  if child_node.state.state not in self.reached.keys() or child_node.path_cost < self.reached[child_node.state.state].path_cost:
213                      self.reached[child_node.state.state] = child_node
214                      self.frontier.add(child_node)
215          self.finish(False, None)
216  
217      def expand(self, node: SearchNode) -> set[SearchNode]:
218          expanded: set = set()
219          for _, child_id, edge_data in node.state.edges:
220              path_cost = node.path_cost + edge_data['weight']
221              child_node = SearchNode(StateNode(child_id, self.graph.edges(child_id, data=True)), node, path_cost)
222              expanded.add(child_node)
223          return expanded
224  
225      def finish(self, success: bool, last_node):
226          if success:
227              print(f'success!')
228              print(f'reached goal node: {last_node.state.state}')
229              print(f"goal node cost: {last_node.path_cost}")
230              solution_path: list[Hashable] = []
231              path_node = last_node
232              reached_start = False
233              while not reached_start:
234                  solution_path.append(path_node.state.state)
235                  if not path_node.parent:
236                      reached_start = True
237                  else:
238                      path_node = path_node.parent
239              solution_path.reverse()
240              print(f"path to goal: {solution_path}")
241          else:
242              print(f'finished without success.')
243              print(f"reached nodes:")
244              for key, value in self.reached.items():
245                  print(f"{key}, cost: {value.path_cost}")
